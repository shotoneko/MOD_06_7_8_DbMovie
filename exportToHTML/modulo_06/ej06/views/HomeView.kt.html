79   @Composable
80   fun ContentHomeView(
81       paddingValues: PaddingValues,
82       viewModel: HomeViewVM
83   ) {
84       val movies by viewModel.movies.collectAsState(listOf())
85       val isLoading by viewModel.isLoading.collectAsState(false)
86   
87       LazyColumn(
88           modifier = Modifier
89               .fillMaxSize()
90               .padding(paddingValues)
91               .padding(8.dp)
92       ) {
93           items(movies) { movie ->
94               MovieCard(movie = movie, onDeleteClick = { viewModel.deleteMovie(movie) })
95           }
96           // Loading indicator at the bottom
97           item {
98               if (isLoading) {
99                   Box(modifier = Modifier.fillMaxWidth().wrapContentSize(Alignment.Center)) {
100                      CircularProgressIndicator()
101                  }
102              }
103          }
104      }
105  }

107  @Composable
108  fun MovieCard(movie: DbMovie, onDeleteClick: () -> Unit) {
109      Card(
110          shape = RoundedCornerShape(8.dp),
111          elevation = CardDefaults.cardElevation(2.dp),
112          colors = CardDefaults.cardColors(containerColor = Color.White),
113          modifier = Modifier
114              .padding(horizontal = 8.dp, vertical = 8.dp)
115              .fillMaxWidth()
116      ) {
117          Row(
118              modifier = Modifier
119                  .padding(16.dp)
120          ) {
121              Column(
122                  modifier = Modifier.weight(1f)
123              ) {
124                  Text(movie.title, style = TextStyle(fontWeight = FontWeight.Bold))
125                  Text(movie.overview ?: "") // Handle null overview
126              }
127              Spacer(modifier = Modifier.width(16.dp))
128              IconButton(
129                  onClick = onDeleteClick
130              ) {
131                  Icon(Icons.Default.Delete, "Delete")
132              }
133          }
134      }
135  }
